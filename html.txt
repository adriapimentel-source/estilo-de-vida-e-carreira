function goTo(n) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  const id = n === 'result' ? 'page-result' : 'page-' + n;
  const el = document.getElementById(id);
  if (el) { el.classList.add('active'); window.scrollTo({top:0, behavior:'smooth'}); }
}

function iniciar() {
  let ok = true;
  const nome  = v('nome');
  const ra    = v('ra');
  const turma = v('turma');
  if (!nome)  { markErr('nome','err-nome');  ok=false; } else clearErr('nome','err-nome');
  if (!ra)    { markErr('ra','err-ra');      ok=false; } else clearErr('ra','err-ra');
  if (!turma) { markErr('turma','err-turma'); ok=false; } else clearErr('turma','err-turma');
  if (ok) goTo(1);
}

function avancar(step) {
  const checks = {
    1: () => {
      let ok = true;
      if (!v('cidade-lugar'))  { markErr('cidade-lugar','err-cidade');  ok=false; } else clearErr('cidade-lugar','err-cidade');
      if (!v('desc-moradia'))  { markErr('desc-moradia','err-moradia'); ok=false; } else clearErr('desc-moradia','err-moradia');
      return ok;
    },
    2: () => {
      let ok = true;
      if (!v('porque-area'))  { markErr('porque-area','err-area');   ok=false; } else clearErr('porque-area','err-area');
      if (!v('valores-trab')) { markErr('valores-trab','err-valores'); ok=false; } else clearErr('valores-trab','err-valores');
      return ok;
    },
    3: () => {
      let ok = true;
      if (!v('porque-renda')) { markErr('porque-renda','err-renda'); ok=false; } else clearErr('porque-renda','err-renda');
      return ok;
    },
    4: () => {
      let ok = true;
      if (!v('desc-familiar'))  { markErr('desc-familiar','err-familiar');   ok=false; } else clearErr('desc-familiar','err-familiar');
      if (!v('qual-relacoes')) { markErr('qual-relacoes','err-relacoes'); ok=false; } else clearErr('qual-relacoes','err-relacoes');
      return ok;
    },
    5: () => true,
  };
  if (checks[step] && !checks[step]()) return;
  if (step === 5) buildRevSliders();
  goTo(step + 1);
}

function markErr(fieldId, errId) {
  const f = document.getElementById(fieldId);
  const e = document.getElementById(errId);
  if (f) f.style.borderColor = 'var(--danger)';
  if (e) e.style.display = 'block';
}
function clearErr(fieldId, errId) {
  const f = document.getElementById(fieldId);
  const e = document.getElementById(errId);
  if (f) f.style.borderColor = '';
  if (e) e.style.display = 'none';
}

// RODA DA VIDA — Canvas responsivo dentro do retângulo
// ──────────────────────────────────────────────
function desenharRoda() {
  const canvas = document.getElementById('roda-canvas');
  if (!canvas) return;

  // Medir o container real após o layout
  const wrap = canvas.parentElement;
  const size  = wrap.clientWidth;   // sempre igual ao container

  // Resolução real do canvas = tamanho CSS (sem DevicePixelRatio para simplificar)
  canvas.width  = size;
  canvas.height = size;

  const ctx = canvas.getContext('2d');
  const cx = size / 2;
  const cy = size / 2;

  /* Layout fractions:
     - Wheel radius      : 36% of size  →  labels end at ~46%
     - Label radius      : 46% of size
     Together = 92% of half = ≈46% × 2 = 92% of size → fits with 4% margin each side */
  const maxR   = size * 0.355;
  const labelR = size * 0.455;
  const n       = CATS.length;         // 12
  const step    = (2 * Math.PI) / n;
  const start   = -Math.PI / 2;        // começa do topo

  ctx.clearRect(0, 0, size, size);

  // ── Anéis de grade ──────────────────────────
  for (let ring = 1; ring <= 10; ring++) {
    ctx.beginPath();
    ctx.arc(cx, cy, maxR * ring / 10, 0, 2 * Math.PI);
    ctx.strokeStyle = ring === 10 ? '#bbb' : '#ebebeb';
    ctx.lineWidth   = ring === 10 ? 1.5 : 0.6;
    ctx.stroke();
  }

  // ── Spokes ──────────────────────────────────
  for (let i = 0; i < n; i++) {
    const a = start + i * step;
    ctx.beginPath();
    ctx.moveTo(cx, cy);
    ctx.lineTo(cx + maxR * Math.cos(a), cy + maxR * Math.sin(a));
    ctx.strokeStyle = '#ddd';
    ctx.lineWidth   = 0.6;
    ctx.stroke();
  }

  // ── Setores coloridos ────────────────────────
  for (let i = 0; i < n; i++) {
    const a1 = start + i * step;
    const a2 = start + (i + 1) * step;
    const r  = maxR * scores[CATS[i].key] / 10;
    if (r < 1) continue;
    ctx.beginPath();
    ctx.moveTo(cx, cy);
    ctx.arc(cx, cy, r, a1, a2);
    ctx.closePath();
    ctx.fillStyle   = CATS[i].color + '3a';
    ctx.fill();
    ctx.strokeStyle = CATS[i].color;
    ctx.lineWidth   = 1.5;
    ctx.stroke();
  }

  // ── Polígono externo ─────────────────────────
  ctx.beginPath();
  for (let i = 0; i < n; i++) {
    const a = start + i * step;
    const r = maxR * scores[CATS[i].key] / 10;
    const x = cx + r * Math.cos(a);
    const y = cy + r * Math.sin(a);
    i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
  }
  ctx.closePath();
  ctx.strokeStyle = 'rgba(0,0,0,0.12)';
  ctx.lineWidth   = 1;
  ctx.stroke();

  // ── Pontos nas pontas ────────────────────────
  const dotR = Math.max(3, size * 0.012);
  for (let i = 0; i < n; i++) {
    const a = start + i * step;
    const r = maxR * scores[CATS[i].key] / 10;
    ctx.beginPath();
    ctx.arc(cx + r * Math.cos(a), cy + r * Math.sin(a), dotR, 0, 2 * Math.PI);
    ctx.fillStyle = CATS[i].color;
    ctx.fill();
  }

  // ── Labels ───────────────────────────────────
  // FontSize adaptado ao tamanho do canvas
  const fs = Math.max(8, Math.min(11, Math.floor(size * 0.0215)));
  ctx.font = `700 ${fs}px 'Plus Jakarta Sans', sans-serif`;

  for (let i = 0; i < n; i++) {
    const angle = start + i * step;
    const lx    = cx + labelR * Math.cos(angle);
    const ly    = cy + labelR * Math.sin(angle);

    // Normalizar ângulo relativo ao topo (0°= cima, horário)
    const deg = (((angle - start) * 180 / Math.PI) + 360) % 360;

    // Alinhamento baseado em octante (12 itens = 30° cada)
    if (deg < 15 || deg >= 345) {
      ctx.textAlign    = 'center'; ctx.textBaseline = 'bottom';
    } else if (deg < 75) {
      ctx.textAlign    = 'left';   ctx.textBaseline = 'bottom';
    } else if (deg < 105) {
      ctx.textAlign    = 'left';   ctx.textBaseline = 'middle';
    } else if (deg < 165) {
      ctx.textAlign    = 'left';   ctx.textBaseline = 'top';
    } else if (deg < 195) {
      ctx.textAlign    = 'center'; ctx.textBaseline = 'top';
    } else if (deg < 255) {
      ctx.textAlign    = 'right';  ctx.textBaseline = 'top';
    } else if (deg < 285) {
      ctx.textAlign    = 'right';  ctx.textBaseline = 'middle';
    } else {
      ctx.textAlign    = 'right';  ctx.textBaseline = 'bottom';
    }

    ctx.fillStyle = CATS[i].color;
    ctx.fillText(CATS[i].label, lx, ly);
  }

  // ── Texto central ─────────────────────────────
  ctx.textAlign    = 'center';
  ctx.textBaseline = 'middle';
  ctx.font         = `600 ${Math.max(8,Math.floor(fs * 0.9))}px 'Plus Jakarta Sans', sans-serif`;
  ctx.fillStyle    = '#bbb';
  ctx.fillText('2045', cx, cy);
}

// GOOGLE SHEETS
// ──────────────────────────────────────────────
async function enviarSheets() {
  if (SHEETS_URL.includes('COLE_AQUI')) {
    console.warn('[Sheets] URL não configurada. Dados não enviados.');
    return;
  }

  const statusEl = document.getElementById('status-msg');
  statusEl.textContent = '⏳ Enviando respostas...';
  statusEl.className   = 'status ok';

  const payload = {
    timestamp:       new Date().toLocaleString('pt-BR'),
    nome:            v('nome'),
    ra:              v('ra'),
    turma:           v('turma'),
    // Etapa 1
    pais:            getPais(),
    cidadeLugar:     v('cidade-lugar'),
    tipoMoradia:     checked('moradia'),
    descMoradia:     v('desc-moradia'),
    // Etapa 2
    area:            v('area'),
    porqueArea:      v('porque-area'),
    modeloTrab:      checked('modelo-trab'),
    ambTrab:         checked('amb-trab'),
    valoresTrab:     v('valores-trab'),
    // Etapa 3
    renda:           document.getElementById('v-renda').textContent,
    porqueRenda:     v('porque-renda'),
    estiloFin:       checked('estilo-fin'),
    oQuePermite:     v('o-que-permite'),
    // Etapa 4
    vidaAfetiva:     checked('vida-afetiva'),
    descFamiliar:    v('desc-familiar'),
    redeSocial:      checked('rede-social'),
    qualRelacoes:    v('qual-relacoes'),
    // Etapa 5
    saudeCorpo:      checked('saude-corpo'),
    espiritualidade: checked('espiritualidade'),
    lazer:           checked('lazer'),
    contribuicao:    v('contribuicao'),
    // Notas
    nota_plenitude:      scores.plenitude,
    nota_proposito:      scores.proposito,
    nota_intelectual:    scores.intelectual,
    nota_financas:       scores.financas,
    nota_familia:        scores.familia,
    nota_amor:           scores.amor,
    nota_social:         scores.social,
    nota_saude:          scores.saude,
    nota_emocional:      scores.emocional,
    nota_espiritualidade:scores.espiritualidade,
    nota_lazer:          scores.lazer,
    nota_contribuicao:   scores.contribuicao,
  };

  try {
    // Usar text/plain evita preflight CORS com Apps Script
    const res  = await fetch(SHEETS_URL, {
      method: 'POST',
      body:   JSON.stringify(payload),
      headers:{ 'Content-Type': 'text/plain;charset=utf-8' }
    });
    const data = await res.json();
    if (data.success) {
      statusEl.textContent = '✅ Respostas salvas com sucesso no Google Sheets!';
    } else {
      throw new Error(data.error || 'Erro desconhecido');
    }
  } catch(err) {
    statusEl.textContent = '⚠️ Não foi possível enviar automaticamente. Salve o PDF abaixo como backup.';
    statusEl.className   = 'status err';
    console.error('[Sheets]', err);
  }
}
